# Gestor de maquinas virtuales

# sacado de https://www.youtube.com/watch?v=KsN1fjP1pXA para cachy OS

'''
Virt-Manager:
sudo pacman -Syu --needed virt-manager qemu-desktop libvirt edk2-ovmf dnsmasq vde2 bridge-utils openbsd-netcat iptables-nft libguestfs dmidecode virt-viewer

sudo systemctl enable --now libvirtd.service

sudo systemctl start libvirtd.service

sudo virsh net-autostart default

sudo nano /etc/libvirt/libvirtd.conf

uncomment

unix_sock_group = 'libvirt'

unix_sock_rw_perms = '0770'

sudo usermod -a -G libvirt $(whoami)

sudo nano /etc/libvirt/qemu.conf
uncomment
user = "your user name"
group = "your user name"

sudo nano /etc/libvirt/network.conf
firewall_backend = "iptables"
sudo systemctl restart libvirtd
'''

sudo pacman -S swtpm
